Game.java 
import java.util.Scanner;
import java.util.Random;

// Class containing all the logic for the Hangman game
public class Game {
    
    // Setting up variables with private access modifiers 
    private String[] wordList = {"DEFORESTATION", "CLIMATE CHANGE", "OVERFISHING", "ANIMAL RIGHTS", "WATER POLLUTION"}; 
    private String secretWord;
    private char[] guessedWord;
    private int attempts;
    private Scanner scanner;

    // Constructor to set up the initial state of the game
    public Game() {
        Random rand = new Random();
        
        // Randomly select a word from the list every time the game runs
        secretWord = wordList[rand.nextInt(wordList.length)];
        guessedWord = new char[secretWord.length()];
        
        // Fill the hidden word with underscores, but keep spaces visible
        for (int i = 0; i < guessedWord.length; i++) {
            if (secretWord.charAt(i) == ' ') {
                guessedWord[i] = ' ';
            } else {
                guessedWord[i] = '_';
            }
        }

        attempts = 5; 
        scanner = new Scanner(System.in);
    }

    // Method to handle the main game loop
    public void play() {
        System.out.println("Welcome to Eco-Hangman!");

        // Loop while the player has attempts left and there are still underscores
        while (attempts > 0 && new String(guessedWord).contains("_")) {
            displayStatus();
            char userGuess = getValidInput();
            processGuess(userGuess);
        }

        // Show winning or losing message
        if (!new String(guessedWord).contains("_")) {
            System.out.println("\nCongrats! You guessed the word: " + secretWord);
        } else {
            System.out.println("\nGame Over! The word was: " + secretWord);
        }
        
        scanner.close();
    }

    // Method to show current progress and attempts remaining
    private void displayStatus() {
        System.out.println("\nWord: " + new String(guessedWord));
        System.out.println("Attempts left: " + attempts);
    }

    // Method to get input and handle exceptions without crashing
    private char getValidInput() {
        while (true) {
            System.out.print("Guess a letter: ");
            try {
                String input = scanner.nextLine().toUpperCase();
                
                // Check for empty input
                if (input.isEmpty()) {
                    throw new Exception("Input cannot be empty. Please enter a letter.");
                }
                
                char guess = input.charAt(0); 
                
                // Check if the input is actually a letter
                if (!Character.isLetter(guess)) {
                    throw new Exception("Invalid input. Please enter letters only.");
                }
                
                // Return the valid letter
                return guess; 
                
            } catch (Exception e) {
                // Catching any unexpected errors and prompting the user again
                System.out.println("Error: " + e.getMessage());
            }
        }
    }

    // Method to check if the guessed letter is inside the secret word
    private void processGuess(char guess) {
        boolean found = false;
        
        // Go through the secret word to find matching letters
        for (int i = 0; i < secretWord.length(); i++) {
            if (secretWord.charAt(i) == guess) {
                guessedWord[i] = guess; 
                found = true;
            }
        }
        
        // Decrease attempts if the guess was wrong
        if (!found) {
            attempts--;
            System.out.println("Incorrect guess!");
        } else {
            System.out.println("Good guess!");
        }
    }
}
